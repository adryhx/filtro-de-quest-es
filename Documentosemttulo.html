                <input type="text" id="keyword" name="keyword" placeholder="Digite uma palavra-chave">

            <!-- As questões filtradas aparecerão aqui -->

    <script>

        const questions = [

            { id: 1, subject: 'matematica', difficulty: 'facil', text: 'Quanto é 2 + 2?' },

            { id: 2, subject: 'portugues', difficulty: 'medio', text: 'Qual é a função do predicado?' },

            { id: 3, subject: 'ciencias', difficulty: 'dificil', text: 'Explique a teoria da evolução.' },

            { id: 4, subject: 'historia', difficulty: 'facil', text: 'Quem descobriu o Brasil?' },

            { id: 5, subject: 'geografia', difficulty: 'medio', text: 'Quais são os biomas brasileiros?' }

        ];

        document.getElementById('filter-form').addEventListener('submit', function(e) {

            e.preventDefault();

            const subject = document.getElementById('subject').value;

            const difficulty = document.getElementById('difficulty').value;

            const keyword = document.getElementById('keyword').value.toLowerCase();

            const filteredQuestions = questions.filter(q => {

                return (

                    (subject === '' || q.subject === subject) &&

                    (difficulty === '' || q.difficulty === difficulty) &&

                    (keyword === '' || q.text.toLowerCase().includes(keyword))

                );

            });

            const questionList = document.getElementById('question-list');

            questionList.innerHTML = '';

            if (filteredQuestions.length > 0) {

                filteredQuestions.forEach(q => {

                    const div = document.createElement('div');

                    div.className = 'question';

                    div.textContent = q.text;

                    questionList.appendChild(div);

                });

            } else {

                questionList.textContent = 'Nenhuma questão encontrada.';

            }

        });

    </script>
